Create the Pipeline job in Jenkins

Open your Jenkins web UI and sign in.

Click New Item (top-left).

Enter a name (e.g. simple-pipeline) and select Pipeline, then OK.

In the job configuration page, scroll to Pipeline section.

Choose Definition: Pipeline script (this keeps the script inside Jenkins).

Paste the pipeline script (see next block) into the Script text area.

pipeline { agent any; stages {
  stage('Build'){ steps { echo 'BUILD' } }
  stage('Test'){ steps { echo 'TEST' } }
  stage('Deploy'){ steps { echo 'DEPLOY' } }
}}

import java.util.*;
public class calculator{
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        while(true){
            System.out.println("Enter the operation");
            System.out.println("1.addition");
            System.out.println("2.subtraction");
            System.out.println("3.multiplication");
            System.out.println("4.division");


            int ch = sc.nextInt();
            if(ch == 5){
                System.out.println("goodbye");
                break;
            }

            System.out.println("Enter the first Number");
            double a = sc.nextDouble();
            System.out.println("Enter the second Number");
            double b = sc.nextDouble();


            switch(ch){
                case 1 : System.out.println(a+b);break;
                case 2 : System.out.println(a-b);break;
                case 3 : System.out.println(a*b);break;
                case 4 : System.out.println(a/b);break;
                default : System.out.println("invalid option");
            }

        }


    }
}

docker build -t myimage:1.0 .

docker images

docker run -d --name mynginx -p 8080:80 nginx:alpine

docker ps

docker stop mynginx
docker rm mynginx

# show logs
docker logs -f mynginx

docker exec -it mynginx sh

docker rmi myimage:1.0

docker run -it --rm alpine sh

function calc(op) {
  const a = Number(document.getElementById('num1').value);
  const b = Number(document.getElementById('num2').value);
  let res;

  if (op === '+') res = a + b;
  else if (op === '-') res = a - b;
  else if (op === '*') res = a * b;
  else res = a / b;

  document.getElementById('result').textContent = "Result: " + res;
}
</script>

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.edge.EdgeDriver;

public class TestApp {
    public static void main(String[] args) {

        // Set path to msedgedriver.exe if not in system PATH
        System.setProperty("webdriver.edge.driver", "C:\\path\\to\\msedgedriver.exe");

        WebDriver driver = new EdgeDriver();

        try {
            // Open the containerized app
            driver.get("http://localhost:8082");

            // Check page title
            String title = driver.getTitle();
            if(title.isEmpty()) {
                System.out.println("Title should not be empty");
            } else {
                System.out.println("Page title: " + title);
            }

            // Fill name field
            WebElement nameInput = driver.findElement(By.id("name"));
            nameInput.sendKeys("Test User");

            // Click the order button
            WebElement btn = driver.findElement(By.cssSelector("button"));
            btn.click();

            System.out.println("Test completed successfully.");

        } catch(Exception e) {
            e.printStackTrace();
        } finally {
            // Close browser
            driver.quit();
        }
    }
}
