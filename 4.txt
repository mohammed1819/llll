import 'package:flutter/material.dart';

void main() => runApp(MaterialApp(home: Screen1()));

class Screen1 extends StatelessWidget {
  @override
  Widget build(BuildContext c) {
    return Scaffold(
      body: Center(
        child: ElevatedButton(
          onPressed: () => Navigator.push(c, MaterialPageRoute(
            builder: (_) => Screen2(),
          )),
          child: Text("Go"),
        ),
      ),
    );
  }
}

class Screen2 extends StatelessWidget {
  @override
  Widget build(BuildContext c) => Scaffold(body: Center(child: Text("Screen 2")));
}

import 'package:flutter/material.dart';

void main() => runApp(MaterialApp(
  routes: {'/': (_) => Home(), '/next': (_) => Next()},
));

class Home extends StatelessWidget {
  @override
  Widget build(BuildContext c) {
    return Scaffold(
      body: Center(
        child: ElevatedButton(
          onPressed: () => Navigator.pushNamed(c, '/next'),
          child: Text("Next"),
        ),
      ),
    );
  }
}

class Next extends StatelessWidget {
  @override
  Widget build(BuildContext c) => Scaffold(body: Center(child: Text("Named Route")));
}
