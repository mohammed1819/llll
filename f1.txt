3b) Implement media queries and breakpoints for responsiveness.
import 'package:flutter/material.dart';
void main() => runApp(MyApp());
class MyApp extends StatelessWidget {
 @override
 Widget build(BuildContext context) {
 return MaterialApp(home: ResponsiveLayout());
 }
}
class ResponsiveLayout extends StatelessWidget {
 Widget _box(Color c) => Container(color: c, width: 100, height: 100);
 @override
 Widget build(BuildContext context) {
 double w = MediaQuery.of(context).size.width;
 if (w < 600) {
 return Scaffold(
 appBar: AppBar(title: Text('Mobile Layout')),
 body: Column(
 mainAxisAlignment: MainAxisAlignment.center,
 children: [_box(Colors.red), _box(Colors.green), 
_box(Colors.blue)],
 ),
 );
 } else if (w < 1200) {
 return Scaffold(
 appBar: AppBar(title: Text('Tablet Layout')),
 body: Row(
 mainAxisAlignment: MainAxisAlignment.center,
 children: [_box(Colors.red), _box(Colors.green), 
_box(Colors.blue)],
 ),
 );
 } else {
 return Scaffold(
 appBar: AppBar(title: Text('Desktop Layout')),
 body: GridView.count(
 crossAxisCount: 3,
 mainAxisSpacing: 10,
 crossAxisSpacing: 10,
 children: [_box(Colors.red), _box(Colors.green), 
_box(Colors.blue)],
 ),
 );
 }
 }
}
